\begin{enumerate}
	\item Indiqueu el valor dels paràmetres, per tal que les matrius siguin l'una la transposta de l'altra:


$$
\begin{pmatrix}
2 & 4 & c \\
3 & -1 & 4\\
5 & 0 & 2
\end{pmatrix}
\qquad
\begin{pmatrix}
2 & a & b \\
4 & -1 & 0\\
4 & 4 & 2
\end{pmatrix}
$$



\item Escriu una matriu d'ordre 3 que coincideixi amb la seva transposta. En general, com ha de ser una matriu perquè això passi?

\item Donades les matrius següents:

$$
A=
\begin{pmatrix}
-2 & 3 & 1 \\
4 & 5 & 2\\
0 & 3 & -2
\end{pmatrix}
\qquad
B=
\begin{pmatrix}
4 & -2 & 3 \\
-2 & 3 & -1\\
2 & 1 & -3
\end{pmatrix}
$$

Calculeu:

\begin{enumerate}
	\item $3A-B$
	\item $4(A+2B)-3(A-B)$
\end{enumerate}


\item Troba els valors dels paràmetres de manera que $A+C=2B$

$$
A=
\begin{pmatrix}
3 & a & 2 \\
-1 & 2 & c
\end{pmatrix}
\qquad
B=
\begin{pmatrix}
4 & -1 & 3 \\
5 & -2 & 1
\end{pmatrix}
\qquad
C=
\begin{pmatrix}
5 & 2 & b \\
d & e & 4
\end{pmatrix}
$$

\item Calculeu el producte de les següents matrius sempre que sigui possible:

\begin{enumerate}
	\item 
	
	$$
	\begin{pmatrix}
	2 & 3 \\
	1 & -2
	\end{pmatrix}
	\cdot
	\begin{pmatrix}
	4 & -1\\
	0 & 3
	\end{pmatrix}
	$$
	
	\item 
	
	$$
	\begin{pmatrix}
	6 & -2 \\
	2 & -3
	\end{pmatrix}
	\cdot
	\begin{pmatrix}
	4 & 3 & -2\\
	0 & -1 & 3
	\end{pmatrix}
	$$
	
	\item 
	
	$$
	\begin{pmatrix}
	4 & 2 \\
	2 & 3 \\
	1 & 3
	\end{pmatrix}
	\cdot
	\begin{pmatrix}
	3 & -1 & 4\\
	5 & 2 & 0
	\end{pmatrix}
	$$
	
	\item 
	
	$$
	\begin{pmatrix}
	4 & 3 & -2 \\
	2 & 1& 3
	\end{pmatrix}
	\cdot
	\begin{pmatrix}
	5 & 1\\
	4 & -3 \\
	-2 & 1
	\end{pmatrix}
	$$
	
	\item 
	
	$$
	\begin{pmatrix}
	4 & 1 \\
	1 & -2
	\end{pmatrix}
	\cdot
	\begin{pmatrix}
	3 & 1\\
	4 & 5\\
	0 & 3
	\end{pmatrix}
	$$
	
\end{enumerate}



\end{enumerate}
